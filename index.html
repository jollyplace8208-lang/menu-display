function generateDisplayMenu() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let display = ss.getSheetByName("Display_Menu"); if (!display) display = ss.insertSheet("Display_Menu"); display.clear(); // Header display.appendRow([ "日期","早餐","午餐","下午茶","晚餐","湯","戒口" ]); ss.getSheets().forEach(sheet => { if (!sheet.getName().toLowerCase().includes("week")) return; const data = sheet.getDataRange().getValues(); // helper：按 column 取餐 const get = (label, col) => { const row = data.find(r => String(r[0]).trim() === label); return row ? row[col] : ""; }; // 全表掃描日期 for (let r = 0; r < data.length; r++) { for (let c = 0; c < data[r].length; c++) { const cell = data[r][c]; // Excel 日期序號 if (typeof cell === "number" && cell >= 40000 && cell <= 50000) { const date = new Date(Math.round((cell - 25569) * 86400 * 1000)); const dateStr = Utilities.formatDate( date, "Asia/Hong_Kong", "yyyy-MM-dd" ); display.appendRow([ dateStr, get("早餐", c), get("午餐", c), get("下午茶", c), get("晚餐", c), get("湯", c), get("戒口", c) ]); } } } }); }
